class MLP(nn.Module):
    def __init__(self, in_dim, hidden_dim, out_dim):
        super().__init__()
        self.dropout = nn.Dropout(0.2)
        self.flatten = nn.Flatten()
        self.linear_relu_stack = nn.Sequential(
            nn.Linear(in_dim, hidden_dim),
            nn.LeakyReLU(),
            self.dropout,
            nn.Linear(hidden_dim, hidden_dim),
            nn.LeakyReLU(),
            self.dropout,
            nn.Linear(hidden_dim, out_dim),
        )

    def forward(self, x):
        x = self.flatten(x)
        logits = self.linear_relu_stack(x)
        return logits

class CNN(nn.Module):
    def __init__(self, in_size):
        super(CNN, self).__init__()
        self.dropout = nn.Dropout(0.2)
        self.conv1 = nn.Conv2d(in_channels=3, out_channels=8, kernel_size=3, padding=1)
        self.conv2 = nn.Conv2d(in_channels=8, out_channels=16, kernel_size=3, padding=1)
        self.conv3 = nn.Conv2d(in_channels=16, out_channels=16, kernel_size=3, padding=1)
        self.conv4 = nn.Conv2d(in_channels=16, out_channels=32, kernel_size=3, padding=1)
        out_size = in_size // 16
        self.fc1 = nn.Linear(out_size * out_size * 32, 128)
        self.fc2 = nn.Linear(128, 2)

    def forward(self, x):
        x = F.leaky_relu(self.conv1(x))
        x = F.max_pool2d(x, 2)
        x = self.dropout(x)
        x = F.leaky_relu(self.conv2(x))
        x = F.max_pool2d(x, 2)
        x = self.dropout(x)
        x = F.leaky_relu(self.conv3(x))
        x = F.max_pool2d(x, 2)
        x = self.dropout(x)
        x = F.leaky_relu(self.conv4(x))
        x = F.max_pool2d(x, 2)
        x = self.dropout(x)

        x = x.view(x.size(0), -1)
        x = F.leaky_relu(self.fc1(x))
        x = self.fc2(x)
        return x

class RNN(nn.Module):
    def __init__(self, in_dim, h_dim):
        super().__init__()

        self.rnn = nn.RNN(in_dim, h_dim, num_layers=2, bidirectional=True, dropout=0.2)
        self.fc1 = nn.Linear(h_dim, h_dim)

        self.fc2 = nn.Linear(h_dim, 100)
        self.fc3 = nn.Linear(100, 2)

    def forward(self, x):
        x_h = x.permute(2, 0, 1, 3)
        x_h = x_h.reshape(x_h.shape[0], x_h.shape[1], -1)
        x_w = x.permute(3, 0, 1, 2)
        x_w = x_w.reshape(x_w.shape[0], x_w.shape[1], -1)
        x_h = self.compute_rnn(x_h)
        x_w = self.compute_rnn(x_w)
        x = torch.stack([x_h, x_w], dim=0).mean(dim=0)

        x = F.leaky_relu(self.fc2(x))
        x = self.fc3(x)
        return x

    def compute_rnn(self, x):
        _, h_n = self.rnn(x)  # (2*layer, size(batch), h_dim)
        h_n = h_n[-2:]  # (2, size(batch), h_dim)
        h_n = torch.mean(h_n, 0)  # (size(batch), h_dim)

        h_n = self.fc1(h_n)
        h_n = F.leaky_relu(h_n)
        return h_n
